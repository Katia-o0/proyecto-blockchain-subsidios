func (ec *ElegibilidadContract) EvaluarElegibilidad(
    ctx, cuil, tipoSubsidio) {

    // 1. Obtener datos de AFIP desde PDC
    datosAFIP := ec.obtenerDatosPrivadosAFIP(ctx, cuil)
    datosANSES := ec.obtenerDatosPrivadosANSES(ctx, cuil)

    // 2. Obtener parÃ¡metros del subsidio
    parametros := ec.obtenerParametrosSubsidio(ctx, tipoSubsidio)

    // 3. Definir resultado inicial
    resultado := { CUIL: cuil, Elegible: false, Timestamp: Now() }

    // 4. Validar Criterios:
    
    // Criterio 1: Ingresos
    if datosAFIP.IngresoPromedioMensual > parametros.UmbralIngreso {
        resultado.Motivo = "Ingresos superan el lÃ­mite"
        return ec.registrarResultado(ctx, resultado)
    }

    // Criterio 2: Incompatibilidades
    if ec.tieneSubsidioIncompatible(datosANSES, tipoSubsidio) {
        resultado.Motivo = "Ya recibe beneficio incompatible"
        return ec.registrarResultado(ctx, resultado)
    }

    // Criterio 3: ComposiciÃ³n familiar
    if datosANSES.HijosMenores == 0 && 
       !datosANSES.PersonaConDiscapacidad {
        resultado.Motivo = "No cumple criterios familiares"
        return ec.registrarResultado(ctx, resultado)
    }

    // Criterio 4: Patrimonio
    if datosAFIP.ValorBienesRegistrables > 
       parametros.UmbralPatrimonio {
        resultado.Motivo = "Patrimonio supera el lÃ­mite"
        return ec.registrarResultado(ctx, resultado)
    }

    // 5. Calcular nivel y monto
    nivel := ec.calcularNivelSocioeconomico(datosAFIP, datosANSES)
    monto := ec.calcularMontoSubsidio(nivel, parametros)

    // 6. Resultado positivo
    resultado.Elegible = true
    resultado.NivelSocioeconomico = nivel
    resultado.MontoMensual = monto
    resultado.Motivo = "Cumple todos los criterios"

    // 7. Registrar en ledger
    return ec.registrarResultado(ctx, resultado)
}

   * Crea nueva solicitud de subsidio
   */
  async solicitarSubsidio(req, res) {
    try {
      const { cuil, tipoSubsidio, documentos } = req.body;
      
      // 1. Validar CUIL
      if (!this.validarCUIL(cuil)) {
        return res.status(400).json({ 
          error: 'CUIL invÃ¡lido' 
        });
      }

      // 2. Verificar solicitud pendiente
      const solicitudPendiente = 
        await this.solicitudRepo.findPendiente(cuil, tipoSubsidio);
      
      if (solicitudPendiente) {
        return res.status(409).json({ 
          error: 'Ya existe una solicitud pendiente' 
        });
      }

      // 3. Crear registro en base de datos
      const solicitud = await this.solicitudRepo.create({
        cuil,
        tipoSubsidio,
        estado: 'PENDIENTE',
        fechaSolicitud: new Date(),
        documentos: documentos.map(doc => ({
          tipo: doc.tipo,
          hashIPFS: doc.hash,
          fechaCarga: new Date()
        }))
      });

      // 4. Invocar chaincode de elegibilidad (asÃ­ncrono)
      const resultadoPromise = this.fabricService.invokeChaincode(
        'elegibilidad',
        'EvaluarElegibilidad',
        [cuil, tipoSubsidio]
      );

      // 5. Respuesta inmediata al ciudadano
      res.status(202).json({
        message: 'Solicitud en proceso de evaluaciÃ³n',
        solicitudId: solicitud.id,
        tiempoEstimado: '24-48 horas'
      });

      // 6. Procesar resultado en background
      resultadoPromise
        .then(async (resultado) => {
          await this.procesarResultadoElegibilidad(
            solicitud.id, resultado
          );
        })
        .catch(async (error) => {
          await this.manejarErrorEvaluacion(
            solicitud.id, error
          );
        });

    } catch (error) {
      res.status(500).json({ 
        error: 'Error interno del servidor' 
      });
    }
  }

  /**
   * GET /api/subsidios/estado/:cuil
   * Consulta estado de subsidios
   */

      const { cuil } = req.params;

      // 1. Obtener solicitudes de BD

          'ConsultarSubsidiosActivos',
          [cuil]
        );

      const response = {
        solicitudesPendientes: 
          solicitudes.filter(s => s.estado === 'PENDIENTE'),
        subsidiosActivos: subsidiosActivos,
        historial: 
          solicitudes.filter(s => s.estado !== 'PENDIENTE')
      };

      res.json(response);
